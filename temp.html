    # 位移转为数字

    const print = console.log

function toNumber(value) {
  return value >> 0
}

// 无符号位移
function toNumber1(value) {
  return value >>> 0
}

// toNumber(null) // 0
// toNumber({}) // 0
// toNumber("10x") // 0
// toNumber("10") // 10

toNumber(Number.MAX_SAFE_INTEGER) // -1
toNumber1(Number.MAX_SAFE_INTEGER) // 4294967295

    // 无符号
// 解析 toNumber1(Number.MAX_SAFE_INTEGER)
var val = Number.MAX_SAFE_INTEGER.toString(2)
// 11111111111111111111111111111111111111111111111111111
var val1 = val.substring(0, 32)
// 11111111111111111111111111111111
var num = parseInt(val1, 2)
// 4294967295

    大于32位，数字就不准确了。

    // 有符号 最高位为符号位
// 解析 toNumber(Number.MAX_SAFE_INTEGER)

// 十进制变二进制 原码 => 反码 加一（补码）
// 二进制变十进制 减一 => 反码 = 原码

var val = Number.MAX_SAFE_INTEGER.toString(2)
// 11111111111111111111111111111111111111111111111111111
var val1 = val.substring(0, 32)
// 11111111111111111111111111111111
// 减一
// 11111111111111111111111111111110
// 取反
// 00000000000000000000000000000001 = 1
// 最高位是1 表示负数
// -1

    本质：32位的有符号和无符号位移。

    # 数据类型8种判断方式

    # typeof

    操作数的类型，只能识别基础数据类型和引用类型

    typeof null // object
typeof NaN // number
typeof document.all // undefined

    document.all该特性已经从Web标准中删除，虽然一些浏览器目前仍然支持它，会在未来的某个事件停止支持。

    现在typeof已经不是绝对安全的了，因为暂时性死区的问题

    function log() {
  console.log(typeof a)
  let a = 7
}
log() // Uncaught ReferenceError: Cannot access 'a' before initialization

    # constructor

    constructor指向创建实例对象的构造函数。是一种辅助的判断类型手段

    • null和undefined没有这个构造函数
    • constructor可以被改写

    # instanceof

    在原型链上查找，找到即是其实例

    • 右操作数必须是函数或者class
    • 多全局对象，例如多window之间

    # isPrototypeOf

    是否出现在实例对象的原型链上

    • 能正常返回值的情况，基本等同于instanceof
